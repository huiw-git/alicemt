      Select Case RDS.ReadyState
         case 2   'adcReadyStateLoaded
          MsgBox "Executing Query"
         case 3   'adcReadyStateInteractive
          MsgBox "Fetching records in background"
         case 4   'adcReadyStateComplete
          MsgBox "All records fetched"
      End Select

   End Sub
