&lt;SCRIPT Language="VBScript"&gt;

Sub SubmitChange_OnClick
    
    msgbox "Changes will be made"
    RDS.SubmitChanges   
    RDS.Refresh

End Sub

Sub CancelChange_OnClick

    msgbox "Changes will be cancelled"
    RDS.CancelUpdate
    RDS.Refresh

End Sub
